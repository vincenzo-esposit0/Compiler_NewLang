start: def NewLang() : void{
    string check << "si";
    integer op1 << 2;
    integer op2 << 1;
    char op << '+';
    integer result;
    if check = "si" then {
        Stampa("Prova stampa");
        result << Operation(op1, op2, op);
    }
}

def Stampa(string message) : void{
    message << "Ok funziona";
}

def Operation(integer a | integer  b | out string op): integer{
    if op = '+' then {
        return (a+b);
    }
    if op = '-' then {
        return (a-b);
    }
    if op = '*' then {
        return (a*b);
    }
    if op = '/' then {
        return (a/b);
    }
}
